#division x /= y # It is equivalent to x = x / y. divide and assignment operator

x = 20

y = 5

z = 100

x /= y # 20/5=4

print("Result of (x = x / y) = ", x)

Result of (x = x / y) =  4.0

z /= x * y #100/= 4*5=>100/20=5

print("Result = ", z)

Result =  0.25
