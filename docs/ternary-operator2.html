# Nested ternary operator.

num = "num1 = num2" if(num1 == num2) else "num1 > num2" if (num1 > num2) else "num2 > num1"

print(num)

# Nested if-else statement

num1, num2 = 10, 20

if num1 != num2:
    if num1 > num2:
        print("num1 > num2")
    else:
        print("num2 > num1")
else:
    print("num1 = num2")

num2 > num1
