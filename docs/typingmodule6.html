#TypedDict

from typing import TypedDict

class Person(TypedDict):
    name: str
    number: int

p : Person = {"name": "Arnald","age": 22}

print(p["name"],p["age"])

Arnald 22

#Literal type

from typing import Literal

def set_mode(mode: Literal["auto", "manual"]) -> str:
    return f"Mode set to {mode}"

print(set_mode("auto"))

Mode set to auto
