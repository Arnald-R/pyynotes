# bitwise and x &= y # It's equivalent to x = x & y.

x = 20

y = 5

x &= y#=20&5

print("Result of (x = x & y) = ", x)

Result of (x = x & y) =  20

#bitwise or

x = 10

y = 5

x |= y

print("Result of (x = x | y) = ", x)

Result of (x = x | y) =  15

#x ^= y # It's equivalent to x = x ^ y.

x = 20

y = 10

x ^= y

print("Result of (x = x ^ y) = ", x)

Result of (x = x ^ y) =  30

#x >>= y # It's equivalent to x = x >> y.

x = 6

y = 2

x >>= y

print("Result of (x = x >> y) = ", x)

Result of (x = x >> y) =  1

x = 6

y = 2

x <<= y

print("Result of (x = x << y) = ", x)

Result of (x = x << y) =  24
