    return "Hello,Guest"

    def __get__(self,instance,owner):
        return "This is an ReadOnlyDescriptor"

    def __set__(self,instance,value):
        raise AttributeError("cannot define read-only-attribute")

class Myclass:
    read_only = ReadOnlyDescriptor()
