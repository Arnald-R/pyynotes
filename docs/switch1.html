    return div
def multiplication():
    mul = x * y
    return mul
def default():
    return "you selected the wrong operation"

switcher = {
    1 : addition,
    2 : subtraction,
    3 : division,
    4 : multiplication
}

def switch(num):
    return switcher.get(num,default)()
