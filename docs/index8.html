#Using Optional for Nullable Parameters

from typing import Optional

def greet(name: Optional[str] = None) -> str:
    if name:
        return f"Hello,{name}"
    return "Hello,Guest"

print(greet("Arnald"))

Hello,Arnald

print(greet())
