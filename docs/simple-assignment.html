#right-operand  and left-operand

x = 20

y = 5

z = 10

x += y #20 +=5 =>25

print("Result of (x = x + y) = ", x)

Result of (x = x + y) =  25

z += x + y # 10 += 25+5 =>40

print("Result = ", z)

Result =  40
