#accumulating values

def accumulator():
    total = 0
    while True:
        value = yield total
        total += value

coro = accumulator()
next(coro)
print(coro.send(10))
print(coro.send(4))
print(coro.send(24))

from datetime import datetime

def get_age(d):
    d1 = datetime.now()
    months = (d1.year - d.year) * 12 + d1.month - d.month

    year = int(months / 12)
    return year

age = get_age(datetime(1991, 1, 1))

age
