#Connecting Abstract Base class with interfaces

from abc import ABC, abstractmethod

class Flyable(ABC):
    @abstractmethod
    def fly(self):
        pass

class Bird(Flyable):
    def fly(self):
        print("Bird is flying")

bird = Bird()

bird.fly()

Bird is flying
