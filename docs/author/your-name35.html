def fibonacci_coroutine():
    a, b = 0 ,1
    while True:
        yield a
        a, b = a, a + b

coro = fibonacci_coroutine()

for _ in range(6):

def count_up_to(max_val : int) -> Generator[int, None, None]:

    while count <= max_val:
        yield count
        count += 1
