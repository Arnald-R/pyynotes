#Coroutine for producing  Fibanocci Numbers

def fibonacci_coroutine():
    a, b = 0 ,1
    while True:
        yield a
        a, b = a, a + b

coro = fibonacci_coroutine()

for _ in range(6):
    print(next(coro), end =" ")
