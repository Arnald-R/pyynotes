# Implementing switch case statement in Python using Dictionary.

def m():
    return "madurai"
def d():
    return "dharmapuri"
def c():
    return "chengalpet"
def r():
    return "ranipet"
def default():
    return "nocity"

# Creating a dictionary and put these in it.
switcher = {
    1: m,
    2: d,
    3: c,
    4: r
    }

def switch(city):
    return switcher.get(city, default)()

print(switch(2)) # Calling function.
print(switch(3)) # Calling function.
