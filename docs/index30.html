#Abstract Base Class

print("\n === Abstract Base Class ===")

 === Abstract Base Class ===

from abc import ABC, abstractmethod

class Vehicle(ABC):
    @abstractmethod
    def start(self):
        pass

class Car(Vehicle):
    def start(self):
        return "engine started"

car = Car()

print(car.start())

engine started
