#event loop with asyncio.create_task

import asyncio

async def print_numbers():
    for i in range(5):
        print(f"Number:{i}")
        await asyncio.sleep(1)

async def main():
    task = asyncio.create_task(print_numbers())
    print("Task created")
    await asyncio.sleep(3)
    print("Main task comppleted")
    await task

asyncio.run(main())
