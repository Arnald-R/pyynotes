#Abstract Base Class with class methods 

from abc import ABC,abstractmethod

class Database(ABC):
    @classmethod
    @abstractmethod
    def connect(cls):
        pass

class MysqlDatabase(Database):
    @classmethod
    def connect(cls):
        print("connecting to Mysql")

MysqlDatabase.connect()
