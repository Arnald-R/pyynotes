#Preventing dynamic attribute assignment

class Rectangle:
    __slots__ = ['width', 'height']

    def __init__(self,width,height):
        self.width = width
        self.height = height

r = Rectangle(5,10)

try:
    r.color = 'red'
except AttributeError as e:
    print(f"Error: {e}")

Error: 'Rectangle' object has no attribute 'color'
