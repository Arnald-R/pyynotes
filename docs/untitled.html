#using Tuple for fixed length sequences

from typing import Tuple

def process_coordinates(coords: Tuple[float, float]) -> str:
    return f"Latitude: {coords[0]},Latitude: {coords[0]}"

print(process_coordinates((40.7128, -74.0060)))
