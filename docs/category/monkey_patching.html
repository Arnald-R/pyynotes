#Basic Monkey Patching of a function 

#Original Function 
def greet(name):
    return f"Hello, {name}"

#Monkey Patching Function
def new_greet(name):
    return f"Hi, {name}"

greet = new_greet #Replace the original function with the new one

#Test the patched function

print(greet("Arnald"))

Hi, Arnald
