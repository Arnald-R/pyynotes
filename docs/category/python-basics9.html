#Optional Type[Type,None]

from typing import Optional

def greet(name: Optional[str]) -> str:
    if name:
        return f"Hello,{name}!"
    return "Hello, Stranger!"

print(greet("Arnald"))

Hello,Arnald!

print(greet(None))

Hello, Stranger!
