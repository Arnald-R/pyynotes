```python
#Abstract class with multiple methods
```


```python
from abc import ABC, abstractmethod
```


```python
class shape(ABC):
    @abstractmethod
    def area(self):
        pass

    def perimeter(self):
        pass
```


```python
class rectangle(shape):
    def __init__(self, height, width):
        self.height = height
        self.width  = width

    def area(self):
        return self.height * self.width

    def perimeter(self):
        return 2 * (self.height + self.width)
```


```python
rect = rectangle(4,5)
```


```python
print(rect.area())
```

    20
