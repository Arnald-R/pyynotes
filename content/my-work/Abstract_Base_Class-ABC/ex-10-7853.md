```python
#Using Abstract Base class for type checking
```


```python
from abc import ABC,abstractmethod
```


```python
class Media(ABC):
    @abstractmethod
    def play(self):
        pass 
```


```python
class Video(Media):
    def play(self):
        print("playing video")
```


```python
class Audio(Media):
    def play(self):
        print("Playing Audio")
```


```python
def play_Media(media : Media):
    media.play()
```


```python
video =Video()
audio =Audio()
```


```python
play_Media(video)
```

    playing video



```python
play_Media(audio)
```

    Playing Audio
