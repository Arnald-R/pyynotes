```python
#Basic Multithreading with the GIL (CPU-bound)
```


```python
import threading 
import time
```


```python
#CPU Bound task
def cpu_bound_task():
    result = 0
    for i in range(10**7):
        result += i
    print(f"Result: {result}")
```


```python
#start two threads
threads = []
for _ in range(2):
    thread = threading.Thread(target=cpu_bound_task)
    threads.append(thread)
    thread.start()
```
