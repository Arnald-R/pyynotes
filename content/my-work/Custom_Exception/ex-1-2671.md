```python
#Basic Customer Exception
```


```python
class MyCustomError(Exception):
    pass
```


```python
def divide(a,b):
    if b == 0:
        raise MyCustomError("Cannot divide by zero")
        return a/b
```


```python
try:
    result = divide(10,0)
except MyCustomError as e:
    print(f"Error :{e}")
```
