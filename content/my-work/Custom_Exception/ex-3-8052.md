class InvalidErrorException(Exception):
    def __init__(self, value ,code, message = "Invalid value provided"):
        self.value = value
        self.code = code
        self.message = message
        super().__init__(self.message)

    def __str__(self):
        return f"[Error Code {self.code}] {self.message} {self.value}"
