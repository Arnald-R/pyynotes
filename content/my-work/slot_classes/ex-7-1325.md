```python
#Dynamic Behaviour with __slots__
```


```python
class Dynamicperson():
    __slots__ =['name','age']

    def __init__(self,name,age):
        self.name = name
        self.age = age
```


```python
dp = Dynamicperson("Arnald",23)
```


```python
try:
    dp.address ="2/69,poy"
except AttributeError as e:
    print(f"error :{e}")
```
