```python
#Default factory for Mutable Fields
```


```python
from dataclasses import dataclass,field
```


```python
from typing import List
```


```python
@dataclass
class ShippingCart:
    items : List[str] = field(default_factory = list)
```


```python
cart = ShippingCart()
```


```python
cart.items.append("Apple")
```


```python
print(cart)
```
