def file_modes_demo():
    #write in text mode
    with open('modes.txt', 'w', bufferings=1) as f:
        f.write("Line Buffered text\n")

    #Append in binary Mode
    with open('modes.txt', 'ab') as f:
        f.write(b"Binary appended data\n")
