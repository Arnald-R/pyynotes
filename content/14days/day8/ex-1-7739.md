```python
import re
from datetime import datetime
```


```python
text = "Hello, my phone number is 123-456-7890"
```


```python
pattern = r"\d{3}-\d{3}-\d{4}"
```


```python
match = re.search(pattern, text)
```


```python
print(f"Phone number found:{match.group() if match else 'Not found'}")
```
