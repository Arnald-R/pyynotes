```python
class JSONMixin:
    def to_json(self):
        import json
        return json.dumps(self.__dict__)

class Person(JSONMixin):
    def __init__(self, name, age):
        self.name = name
        self.age = age

person = Person("John", 30)
```


```python
print(f"JSON:{person.to_json()}")
```
